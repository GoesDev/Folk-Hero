class createCharacter():


	def __init__(self, character_name, character_class):

    """Inicia a classe criando um personagem"""
        self.name = character_name
        self.class = character_class
        self.experience_points = 0
        self.level = 1
        self.life_points = 0
        self.life_points_max = 0
        self.weapon_damage = 0
        self.inventory = []


	def loseLife(self.life_points, damage_taken):
        """Função usada quando o personagem sofre dano"""
        self.life_points -= damage_taken
        if self.life_points == 0:
            gameOver()


	def gainLife(self.life_points, damage_healed):
        """Função usada quando o personagem cura pontos de vida"""
        self.life_points += damage_healed
        if self.life_points > self.life_points_max:
            self.life_points = self.life_points_max

	def usePotiona(self.life_points):
        """Função usada quando o personagem usa uma poção de cura"""
        self.life_points += 5
        if self.life_points > self.life_points_max:
            self.life_points = self.life_points_max


	def gainXP(self.experience_points, xp_earned):
        """Função usada para ganhar pontos de XP"""
        self.experience_points += xp_earned
        if self.experience_points == 100:
            levelUp()


	def levelUp(self.level):
        """Função usada para subir de nível"""
        self.level += 1
